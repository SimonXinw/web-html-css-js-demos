核心代码实现

/* 基础光泽扫过效果 */ .shine-button { position: relative; overflow: hidden; } .shine-button::before { content: ""; position: absolute; top: 0; left: -100%; width: 100%; height: 100%; background: linear-gradient( 90deg, transparent 0%, rgba(255, 255, 255, 0.3) 50%, transparent 100% ); transition: left 0.6s ease; } .shine-button:hover::before { left: 100%; } /* 自动循环光泽 */ @keyframes shineLoop { 0% { left: -100%; } 50% { left: 150%; } 100% { left: 150%; } } .auto-shine::before { animation: shineLoop 3s infinite; }

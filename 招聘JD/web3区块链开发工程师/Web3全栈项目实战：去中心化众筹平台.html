🚀 Web3 全栈项目实战：去中心化众筹平台

纸上得来终觉浅，绝知此事要躬行。本篇文档将带你从零开始，落地一个完整的 Web3 商业级演示项目——“Web3 众筹平台 (类似去中心化的 Kickstarter)”。在这里，任何人都可以发起筹款目标，任何人都可以使用以太坊 (ETH) 进行捐赠，且资金完全由智能合约自动托管，没有中间商赚差价。

一、 项目完整技术栈介绍

现代 Web3 DApp（去中心化应用）的开发通常被分为“链端（智能合约）”和“客户端（前端）”两部分。我们将采用目前业界最主流的技术栈组合：

⛓️ 智能合约开发

语言: Solidity (^0.8.20)
框架: Hardhat (编译、测试、部署链上代码的利器)
安全库: OpenZeppelin

🖥️ 前端页面开发

框架: React.js / Next.js
UI 库: TailwindCSS
状态管理: React Hooks

🌉 Web3 交互桥梁

核心库: Ethers.js (v6) 或 Wagmi
钱包: MetaMask (小狐狸插件)
RPC: Alchemy / Infura

二、 核心业务逻辑设计

在写代码前，我们要明确合约需要做什么：

  1. 创建项目： 用户调用合约，传入项目标题、描述、目标融资金额和截止时间。合约记录该项目并分配一个唯一 ID。
  2. 捐款（支持项目）： 任何人都可以调用捐款函数，并附带原生的 ETH。合约将资金暂存，并记录谁捐了多少。
  3. 提取资金： 如果达到了众筹时间，且项目发起人想要提款，合约会把池子里的 ETH 自动打入发起人的钱包。

2 编写智能合约 (Crowdfunding.sol)

在 contracts/ 目录下新建 Crowdfunding.sol。这是整个去中心化应用的灵魂，资金的流转全靠它。

pragma solidity ^0.8.20;

contract Crowdfunding {
    // 定义一个项目结构体
    struct Campaign {
        address owner;        // 发起人地址
        string title;         // 项目标题
        uint256 targetAmount; // 目标金额
        uint256 amountCollected;// 已筹集金额
        uint256 deadline;     // 截止时间戳
        bool isClaimed;       // 是否已提取资金
    }

    // 存储所有的项目，用 ID 映射
    mapping(uint256 => Campaign) public campaigns;
    uint256 public numberOfCampaigns = 0; // 项目计数器

    // 事件：用于通知前端发生了什么事
    event CampaignCreated(uint256 id, string title, address owner);
    event Donated(uint256 id, address donator, uint256 amount);

    // 1. 创建项目
    function createCampaign(string memory _title, uint256 _target, uint256 _deadline) public returns (uint256) {
        require(_deadline > block.timestamp, "Deadline must be in the future");

        Campaign storage campaign = campaigns[numberOfCampaigns];
        campaign.owner = msg.sender; // msg.sender 就是调用这个函数的钱包地址
        campaign.title = _title;
        campaign.targetAmount = _target;
        campaign.deadline = _deadline;
        campaign.amountCollected = 0;
        campaign.isClaimed = false;

        emit CampaignCreated(numberOfCampaigns, _title, msg.sender);
        numberOfCampaigns++;
        return numberOfCampaigns - 1;
    }

    // 2. 捐款支持 (payable 关键字表示这个函数可以接收 ETH)
    function donateToCampaign(uint256 _id) public payable {
        uint256 amount = msg.value; // msg.value 是用户随交易附带的 ETH 数量
        Campaign storage campaign = campaigns[_id];

        require(block.timestamp < campaign.deadline, "Campaign ended");
        
        campaign.amountCollected += amount; // 资金自动留存在合约地址中
        emit Donated(_id, msg.sender, amount);
    }

    // 3. 发起人提取资金
    function withdrawFunds(uint256 _id) public {
        Campaign storage campaign = campaigns[_id];
        require(msg.sender == campaign.owner, "Not the owner");
        require(!campaign.isClaimed, "Already claimed");
        require(campaign.amountCollected >= campaign.targetAmount, "Target not reached");

        campaign.isClaimed = true;
        
        // 将合约里的 ETH 发送给 owner (防重入写法)
        (bool sent, ) = payable(campaign.owner).call{value: campaign.amountCollected}("");
        require(sent, "Failed to send Ether");
    }
}

3 本地部署与测试 (Hardhat)

在推向以太坊主网之前，我们必须在本地虚拟链上进行测试。编写 scripts/deploy.js：

const hre = require("hardhat");

async function main() {
  // 获取合约工厂
  const Crowdfunding = await hre.ethers.getContractFactory("Crowdfunding");
  
  // 部署合约
  const crowdfunding = await Crowdfunding.deploy();
  await crowdfunding.waitForDeployment();

  console.log("Crowdfunding 合约已部署至:", await crowdfunding.getAddress());
}

main().catch((error) => {
  console.error(error);
  process.exitCode = 1;
});

运行 npx hardhat node 启动本地链，然后运行 npx hardhat run scripts/deploy.js --network localhost 进行部署。你将得到一个合约地址（例如：0x5FbDB...）。

4 前端 DApp 集成 (React + Ethers.js)

来到 frontend/src/App.js，我们要让前端页面连接 MetaMask 钱包，并调用刚才部署的智能合约。

import React, { useState } from 'react';
import { ethers } from 'ethers';

// 替换为你刚部署的合约地址
const CONTRACT_ADDRESS = "0x5FbDB2315678afecb367f032d93F642f64180aa3";
// 编译合约后生成的 ABI 文件内容摘要
const ABI = [
  "function createCampaign(string _title, uint256 _target, uint256 _deadline) public returns (uint256)",
  "function donateToCampaign(uint256 _id) public payable",
  "function campaigns(uint256) public view returns (address, string, uint256, uint256, uint256, bool)"
];

export default function App() {
  const [account, setAccount] = useState("");
  const [contract, setContract] = useState(null);

  // 1. 连接钱包
  const connectWallet = async () => {
    if (window.ethereum) {
      // 请求 MetaMask 授权
      const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' });
      setAccount(accounts[0]);

      // 初始化 Provider 和 Signer
      const provider = new ethers.BrowserProvider(window.ethereum);
      const signer = await provider.getSigner(); // 获取签名者(用户自己)
      
      // 连接带有签名权限的合约实例
      const crowdfundContract = new ethers.Contract(CONTRACT_ADDRESS, ABI, signer);
      setContract(crowdfundContract);
    } else {
      alert("请安装 MetaMask 钱包插件！");
    }
  };

  // 2. 发起一个众筹项目 (写入链)
  const handleCreateCampaign = async () => {
    if (!contract) return;
    try {
      // 目标 1 个 ETH，时间设置为当前时间 + 7天
      const target = ethers.parseEther("1.0"); 
      const deadline = Math.floor(Date.now() / 1000) + 7 * 24 * 60 * 60;
      
      // 唤起 MetaMask 请求用户签名这笔交易
      const tx = await contract.createCampaign("救救流浪猫项目", target, deadline);
      await tx.wait(); // 等待区块打包确认
      
      alert("项目创建成功！");
    } catch (err) {
      console.error(err);
    }
  };

  // 3. 给 ID 为 0 的项目捐款 ETH
  const handleDonate = async () => {
    if (!contract) return;
    try {
      // 在调用合约函数时，通过 overrides 对象传入 { value: ... } 表示附带的 ETH 数量
      const tx = await contract.donateToCampaign(0, { 
        value: ethers.parseEther("0.1") // 捐赠 0.1 ETH
      });
      await tx.wait();
      
      alert("感谢您的捐赠！0.1 ETH 已转入合约锁仓。");
    } catch (err) {
      console.error(err);
    }
  };

  return (
    <div style={{ padding: "20px" }}>
      <h1>Web3 去中心化众筹</h1>
      {!account ? (
        <button onClick={connectWallet}>🔌 连接钱包</button>
      ) : (
        <div>
          <p>✅ 钱包已连接: {account}</p>
          <button onClick={handleCreateCampaign}>🚀 发起众筹</button>
          <button onClick={handleDonate} style={{ marginLeft: "10px" }}>💰 捐赠 0.1 ETH</button>
        </div>
      )}
    </div>
  );
}

💡 总结与进阶思路

在这个实战项目中，我们走通了 “编写合约 -> 部署合约 -> 前端连接钱包 -> 签名交互” 的全栈闭环。真正做到代码即信任：捐款的钱不经过任何人手，直接进入智能合约地址，达成条件自动解锁提款。

未来的优化方向（作业）：

  • 在合约中增加“退款”功能：如果到了截止日期还没凑够钱，允许捐款人调用 refund() 原路退回 ETH。
  • 前端页面渲染 campaigns 的列表，使用户能直观看到每个项目的进度条。
  • 使用 The Graph 或 Alchemy 替换本地网络，将项目真实部署到 Sepolia 测试网。
